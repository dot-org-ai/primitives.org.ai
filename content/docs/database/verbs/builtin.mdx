# Built-in Verbs

Pre-defined verbs for common operations.

## Importing

```typescript
import { Verbs } from 'ai-database'
```

## create

```typescript
Verbs.create
// {
//   action: 'create',
//   actor: 'creator',
//   act: 'creates',
//   activity: 'creating',
//   result: 'creation',
//   reverse: {
//     at: 'createdAt',
//     by: 'createdBy',
//     in: 'createdIn',
//     for: 'createdFor',
//   },
//   inverse: 'delete',
// }
```

**Usage:**
- Field: `createdAt`, `createdBy`
- Event: `Post.created`
- Actor: `post.creator`
- Label: "Create Post", "Creating..."

## Using Built-in Verbs

### In Schema

```typescript
const db = DB({
  Post: {
    title: 'string',
    content: 'markdown',

    // From Verbs.create
    createdAt: 'datetime',
    createdBy: 'User',

    // From Verbs.update
    updatedAt: 'datetime',
    updatedBy: 'User?',

    // From Verbs.publish
    publishedAt: 'datetime?',
    publishedBy: 'User?',
  },
})
```

### In Nouns

```typescript
const Post = defineNoun({
  singular: 'post',
  plural: 'posts',
  actions: [
    Verbs.create,
    Verbs.update,
    Verbs.delete,
    Verbs.publish,
    Verbs.archive,
  ],
})
```

### For Events

```typescript
db.events.on(`Post.${Verbs.create.action}d`, handler)
db.events.on(`Post.${Verbs.publish.action}ed`, handler)
```

### Field Access

```typescript
// Get timestamp field name
const createdField = Verbs.create.reverse.at  // 'createdAt'
const publishedField = Verbs.publish.reverse.at  // 'publishedAt'

// Get actor field name
const creatorField = Verbs.create.reverse.by  // 'createdBy'
```

### UI Labels

```typescript
function getButtonLabel(verb: Verb): string {
  return verb.action.charAt(0).toUpperCase() + verb.action.slice(1)
}

getButtonLabel(Verbs.publish)  // 'Publish'
getButtonLabel(Verbs.archive)  // 'Archive'
```

### Progress Indicators

```typescript
function getProgressText(verb: Verb): string {
  return `${verb.activity}...`
}

getProgressText(Verbs.publish)  // 'publishing...'
getProgressText(Verbs.create)   // 'creating...'
```

## Extending Built-in Verbs

```typescript
const myPublish = {
  ...Verbs.publish,
  description: 'Make post publicly visible',
  reverse: {
    ...Verbs.publish.reverse,
    on: 'publishedOn',  // custom field
  },
}
```

## Type Safety

```typescript
import { Verb, Verbs } from 'ai-database'

function processVerb(verb: Verb) {
  console.log(verb.action)
  console.log(verb.reverse?.at)
}

processVerb(Verbs.create)
processVerb(Verbs.publish)
```
