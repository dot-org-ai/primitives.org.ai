```bash
npm install @primitives/types
```

## Quick Start

```typescript
import type {
  Employee,
  Invoice,
  Contract,
  Lead
} from '@primitives/types'

// Type-safe business entities
const employee: Employee = {
  id: 'emp-001',
  firstName: 'Jane',
  lastName: 'Doe',
  email: 'jane@company.com',
  departmentId: 'dept-eng',
  status: 'active',
  createdAt: new Date(),
  updatedAt: new Date()
}
```

## Type Pattern

Every domain follows a consistent pattern:

```typescript
// Core type
export interface Employee {
  id: string
  firstName: string
  // ... fields
  createdAt: Date
  updatedAt: Date
}

// Input variant (id and timestamps optional)
export type EmployeeInput = Input<Employee>

// Output variant (full type)
export type EmployeeOutput = Output<Employee>

// CRUD actions
export interface EmployeeActions extends CRUDResource<Employee, EmployeeInput> {
  promote: Action<{ employeeId: string; newTitle: string }, Employee>
  // ... domain-specific actions
}

// Domain events
export interface EmployeeEvents {
  'employee.created': BaseEvent<'employee.created', Employee>
  'employee.promoted': BaseEvent<'employee.promoted', { employee: Employee; previousTitle: string }>
}

// Resource (actions + event subscription)
export interface EmployeeResource extends EmployeeActions {
  on<E extends keyof EmployeeEvents>(event: E, handler: (data: EmployeeEvents[E]) => void): void
}
```

## Domain Proxies

Each module exports a Proxy interface aggregating all resources:

```typescript
import type { HRProxy, FinanceProxy, SalesProxy } from '@primitives/types'

// HR operations
const hr: HRProxy = getHRProxy()
const employee = await hr.employees.create({ firstName: 'Jane', ... })
hr.employees.on('employee.hired', (event) => console.log('New hire:', event.data))

// Finance operations
const finance: FinanceProxy = getFinanceProxy()
const invoice = await finance.invoices.create({ customerId: 'cust-001', ... })

// Sales operations
const sales: SalesProxy = getSalesProxy()
const leads = await sales.leads.list({ status: 'qualified' })
```
